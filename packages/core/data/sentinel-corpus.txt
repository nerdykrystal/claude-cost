export interface EstimateOptions {
  pricingDb?: PricingDatabase;
  asOf?: Date;
  stalenessDays?: number;
}

  const opusGenerations = new Set<string>();

  for (const stage of plan.stages) {
    const pricing = getPricing(stage.model, db);

      opusGenerations.add(pricing.generation);
    }
    const effInputTokens = effectiveInputTokens(stage, plan);
    const cost = estimateStage(stage, pricing, effInputTokens);
    per_stage.push(cost);
  }

  const testPlans = [

  // Validation and error handling
  function validatePlan(plan: Plan): void {
    if (!plan.plan_name) throw new Error("plan_name required");
    if (!Array.isArray(plan.stages)) throw new Error("stages must be array");
    for (const stage of plan.stages) {
      if (!stage.id) throw new Error("stage.id required");
      if (!stage.model) throw new Error("stage.model required");
      if (stage.input_tokens < 0) throw new Error("input_tokens must be >= 0");
    }
  }

  // Tokenizer correction application
  function applyCorrection(tokens: number, correction: number): number {
    return Math.round(tokens * correction);
  }

  // Cost computation helper
  function computeCost(tokens: number, rate: number, correction: number): number {
    const adjusted = applyCorrection(tokens, correction);
    return (adjusted / 1_000_000) * rate;
  }

  // Retry envelope modeling
  function computeEnvelope(subtotal: number, p: number, maxRetries: number): {
    best_case_usd: number;
    expected_cost_usd: number;
    worst_case_usd: number;
  } {
    const best = subtotal;
    const worst = subtotal * (1 + maxRetries);
    const expected = subtotal * (1 + p * maxRetries / (1 - p + 0.0001));
    return { best_case_usd: best, expected_cost_usd: expected, worst_case_usd: worst };
  }

  // Batch processing implementation
  function processBatch(stages: Stage[], pricing: Record<string, ModelPricing>): StageCost[] {
    return stages.map(stage => {
      const p = pricing[stage.model];
      const batchFactor = stage.batch ? 1 - p.batch_discount : 1;
      const correction = p.tokenizer_correction;
      const input = stage.input_tokens * correction * batchFactor * p.input_per_mtok_usd / 1_000_000;
      return { stage_id: stage.id, input_cost_usd: input } as StageCost;
    });
  }
